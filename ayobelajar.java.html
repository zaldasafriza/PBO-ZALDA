package insert;
import java.util.Scanner;

class hitungKalkulator
{
    public double tambah(double nilai_1, double nilai_2)
    {
        double hasil = nilai_1+nilai_2;
        return hasil;
    }
    public double kurang(double nilai_1, double nilai_2)
    {
        double hasil = nilai_1-nilai_2;
        return hasil;
    }
    public double kali(double nilai_1, double nilai_2)
    {
        double hasil = nilai_1*nilai_2;
        return hasil;
    }

}
public class demoKonversi
{
    public static void main(String[] args)
    {
        // pembentukan objek
        hitungKalkulator obj = new hitungKalkulator();
        Scanner sc = new Scanner(System.in);

        System.out.println("Masukan nilai 1 : ");
        double nilai_1 = sc.nextDouble();

        System.out.println("Masukan nilai 2 : ");
        double nilai_2 = sc.nextDouble();
        System.out.println("Hasil Tambah : "+obj.tambah(nilai_1, nilai_2));
        System.out.println("Hasil Kali : "+obj.kali(nilai_1, nilai_2));
        System.out.println("Hasil Kurang : "+obj.kurang(nilai_1, nilai_2));

    }
}
